# Commands 

## Set Gain
gain_cmd=f'/usr/bin/amixer sset ADC {gain}db'
gain_process = subprocess.Popen(arecord_cmd.split(), stdout=subprocess.PIPE, stderr=subprocess.PIPE)
gain_stdout,gain_stderr=gain_proc.communicate()

## Record Audio

rec_cmd=f'arecord -q -D sysdefault -r 44100 -d {self.rec_time} -f S16 -V mono {self.savedest}'
rec_process = subprocess.Popen(rec_cmd.split(), stdout=subprocess.PIPE, stderr=subprocess.PIPE)
rec_stdout,rec_stderr=arecord_proc.communicate()

## Post processing task

ffmpeg_cmd=f'ffmpeg -i {savedest} -y -ac 1 -ar 44100 -sample_fmt s16 {self.proc_dest}'
ffmpeg_process = subprocess.Popen(ffmpeg_cmd.split(), stdout=subprocess.PIPE, stderr=subprocess.PIPE)
ffmpeg_stdout,ffmpeg_stderr=ffmpeg_proc.communicate()
